Timeline.js

Full featured Timeline chart with interactions such as zooming, panning, and scrolling. Dynamic timescale for infinite scrolling. Make it your own through CSS customization.

    const div = document.getElementById('timeline-chart')

    let data = [
        {
          title: 'Lorem ipsum dolor sit amet, blandit fusce aliquet',
          start: new Date(2018, 1, 1),
          end: new Date(2018, 4, 1),
          color: '#65587f'
        },
        {
          title: 'Sed ut perspiciatis, unde omnis iste natus error sit voluptatem',
          start: new Date(2018, 3, 1),
          end: new Date(2018, 7, 1),
          color: '#f18867'
        },
        {
          title: 'Veritatis et quasi architecto beatae vitae dicta sunt, explicabo',
          start: new Date(2018, 2, 15),
          end: new Date(2018, 9, 1),
          color: '#e85f99'
        },
        {
          title: 'At vero eos et accusamus et iusto odio dignissimos ducimus',
          start: new Date(2018, 7, 20),
          end: new Date(2018, 11, 15),
          color: '#50bda1'
        },
    ]

    let onClick = (range) => {
      document.getElementById('clickText').innerHTML = JSON.stringify(range)
    }

    let chart = new timeline.chart(div, data, {
      width: 800, 
      height: 400, 
      rangeHeight: 12, 
      rangeCornerRadius: 4, 
      onClick
    })
